    datacube["ndvi"] = (datacube["nir"] - datacube["red"]) / (
        datacube["nir"] + datacube["red"]
    )

    fig, ax = plt.subplots()
    mean_ndvi = datacube[["ndvi"]].groupby("time").mean(...).to_array(dim="band")

        mean_ndvi.time,
        (mean_ndvi.values + std_ndvi.values)[0, ...],

        alpha=0.3,
        color="C1",
    )
    mean_ndvi.plot(ax=ax, c="C1")
